					ent_hour[i] = (rand() % 15) + 06; // 시간 단위 랜덤 난수 입력 받기 (06시~21시까지 이용가능)
					ent_min[j] = (rand() % 59); // 분 단위 랜덤 난수 입력 받기

				}
			}
			//수정필요. hour범위 6~21 , min범위:00~59

			}
